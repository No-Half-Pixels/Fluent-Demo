<?php
/**
 * Fluent Demo Template class.fluent-demo-template.php.
 *
 * This class provides a container for all settings sent to the demo template file.
 *
 * @package Fluent Demo
 * @since 1.0.0
 * @version 1.0.0
 */
/**
 * Fluent_Demo_Template.
 */
class Fluent_Demo_Template extends Fluent_Base{

	/**
     * @var object $instance holds and instance of the class so we always get the right one.
     */
	private static $instance;

	/**
     * @var array $_args holds the data passed to the class.
     */
	private $_args = array();

	/**
      * Function used to get the class instance
      *
      * @since 1.0.0
      *
      * @return $instance
      *
      */
    public static function getInstance() {

        if (!isset(self::$instance)) {
            self::$instance = new self();
        }

        return self::$instance;
    }

    /**
      * Used to access variables in the class.
      *
      * @var string $key - the key in the data stored
      *
      * @since 1.0.0
      *
      * @return mixed
      *
      */
    public function &__get($key) {
	 return $this->_args[$key];
	}

	/**
      * Used to set variables in the class.
      *
      * @var string $key - the key in the data stored
      * @var string $value - the value to assign
      *
      * @since 1.0.0
      *
      * @return mixed
      *
      */
	public function __set($key, $value) {
		$this->_args[$key] = $value;
	}

	/**
      * Magic method for testing isset again properties.
      *
      * @var string $key - the key in the data stored
      *
      * @since 1.0.0
      *
      * @return boolean
      *
      */
	public function __isset($key){
		return isset($this->_args[$key]);
	}

}
